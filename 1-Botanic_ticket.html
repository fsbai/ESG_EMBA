def plot_moustache_sans_outliers(df, col_name):
    # Supprimer les valeurs extrmes
    df = df[(df[col_name] >= df[col_name].quantile(0.01)) & (df[col_name] <= df[col_name].quantile(0.99))]

    # Tracer le boxplot
    sns.boxplot(x=df[col_name])

    # Ajouter un titre
    plt.title('Courbe de moustache sans les valeurs extrmes de ' + col_name)

    # Afficher le graphique
    plt.show()

# Fonction qui elimine les outliers
def eliminer_outliers(df, col_name, q1=0.01, q2=0.99):
    # Calculer les quantiles
    q_low = df[col_name].quantile(q1)
    q_high = df[col_name].quantile(q2)

    # Supprimer les valeurs en dehors des quantiles
    df = df[(df[col_name] >= q_low) & (df[col_name] <= q_high)]

    # Retourner le dataframe modifi
    return df

# Fonction qui affiche les quantiles d'une variable
def ShowQuantiles(dfsource, col):
    # Calcul des quantiles
    q1 = dfsource[col].quantile(0.25)
    q2 = dfsource[col].quantile(0.5)
    q3 = dfsource[col].quantile(0.75)
    q4 = dfsource[col].quantile(0.95)
    q5 = dfsource[col].quantile(0.99)
    # Cration d'une liste contenant les quantiles
    quantiles = [q1, q2, q3,q4, q5]
    # Cration d'un dataframe avec les quantiles
    df_quantiles = pd.DataFrame({col: quantiles})
    # Affichage du graphique  bar
    df_quantiles.plot.bar()
