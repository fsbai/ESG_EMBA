    df.to_parquet(ParquetPath)

# Fonction d'affichage du graphique pour afficher les valeurs extrêmes
def plot_moustache_sans_outliers(df, col_name):
    # Supprimer les valeurs extrêmes
    df = df[(df[col_name] >= df[col_name].quantile(0.01)) & (df[col_name] <= df[col_name].quantile(0.99))]

    # Tracer le boxplot
    sns.boxplot(x=df[col_name])

    # Ajouter un titre
    plt.title('Courbe de moustache sans les valeurs extrêmes de ' + col_name)

    # Afficher le graphique
    plt.show()

# Fonction qui elimine les outliers
def eliminer_outliers(df, col_name, q1=0.01, q2=0.99):
    # Calculer les quantiles
    q_low = df[col_name].quantile(q1)
    q_high = df[col_name].quantile(q2)

    # Supprimer les valeurs en dehors des quantiles
    df = df[(df[col_name] >= q_low) & (df[col_name] <= q_high)]

    # Retourner le dataframe modifié
    return df

# Fonction qui calcule les scores RFM d'un client : Récence, Fréquence et valeur monétaire 

def calculer_RFM_actif(df, date_col, customer_id_col, transaction_value_col):
    # Calculer la date maximale dans le dataframe
    max_date = df[date_col].max() #datetime.strptime(df[date_col].max(), '%Y-%m-%d %H:%M:%S')

    # Calculer la récence pour chaque client
    #recency_df = df.groupby(customer_id_col)[date_col].apply(lambda x: (max_date - datetime.strptime(x.max(), '%Y-%m-%d  %H:%M:%S')).days)
    recency_df = df.groupby(customer_id_col)[date_col].apply(lambda x: (max_date - x.max()).days)
    recency_df.name = 'recency'

    # Calculer la fréquence pour chaque client
    frequency_df = df.groupby(customer_id_col)[transaction_value_col].count()
    frequency_df.name = 'frequency'

    # Calculer la valeur monétaire pour chaque client
    monetary_value_df = df.groupby(customer_id_col)[transaction_value_col].sum()
    monetary_value_df.name = 'monetary_value'

    # Joindre les résultats en un seul dataframe
    rfm_df = pd.concat([recency_df, frequency_df, monetary_value_df], axis=1)

    # Calculer les scores RFM
    rfm_df['r_score'] = pd.qcut(rfm_df['recency'], q=4, labels=False) + 1
    rfm_df['f_score'] = pd.qcut(rfm_df['frequency'], q=4, labels=False) + 1
    rfm_df['m_score'] = pd.qcut(rfm_df['monetary_value'], q=4, labels=False) + 1

    # Calculer le score RFM
    #rfm_df['rfm_score'] = rfm_df['r_score'].astype(str) + rfm_df['f_score'].astype(str) + rfm_df['m_score'].astype(str)

    # Identifier si le client est actif ou non
    # si les scores de la recence et la frequence et le montant sont superieur à 2 alors le client est actif sinon le client est inactif 
    #rfm_df['TypeClient'] = rfm_df.apply(lambda row: 'actif' if int(row['r_score']) > 2 and int(row['f_score']) > 2 and int(row['m_score']) > 2 else 'inactif', axis=1)

    return rfm_df
